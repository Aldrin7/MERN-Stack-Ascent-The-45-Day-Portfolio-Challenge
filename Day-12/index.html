Essential MongoDB Patterns

Connection & Setup:
const { MongoClient } = require('mongodb');
const client = new MongoClient('mongodb://localhost:27017');
await client.connect();
const db = client.db('myapp');
CRUD Operations:
// Create
await db.collection('items').insertOne(data);

// Read
await db.collection('items').find({}).toArray();

// Update
await db.collection('items').updateOne(
  { _id: new ObjectId(id) },
  { $set: updateData }
);

// Delete
await db.collection('items').deleteOne(
  { _id: new ObjectId(id) }
);

Complete CRUD Implementation

POST - Create:
app.post('/api/items', async (req, res) => {
  try {
    const result = await db.collection('items')
      .insertOne(req.body);
    res.status(201).json({
      success: true,
      data: { ...req.body, _id: result.insertedId }
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      error: 'Failed to create item'
    });
  }
});
GET - Read:
app.get('/api/items', async (req, res) => {
  try {
    const items = await db.collection('items')
      .find({}).toArray();
    res.json({
      success: true,
      count: items.length,
      data: items
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      error: 'Failed to retrieve items'
    });
  }
});

Error Handling Patterns

ObjectId Validation:
const { ObjectId } = require('mongodb');

function isValidObjectId(id) {
  return ObjectId.isValid(id) &&
         new ObjectId(id).toString() === id;
}

// Usage in routes
app.put('/api/items/:id', async (req, res) => {
  if (!isValidObjectId(req.params.id)) {
    return res.status(400).json({
      success: false,
      error: 'Invalid item ID format'
    });
  }
  // Continue with valid ObjectId
});
Not Found Handling:
const result = await db.collection('items')

    { $set: updateData }
  );

if (result.matchedCount === 0) {

  });
